% Setting up the problem and assumptions
\section*{Setup and Assumptions}

% Defining the matrices and vectors
Let $\mathcal{T} = \{1, 2, \dots, T\}$ index time at a monthly frequency. Define the pre-treatment period as $\mathcal{T}_1 \mathrel{\mathop:}= \{t \in \mathcal{T} : t \leq T_0\}$ and the post-treatment period as $\mathcal{T}_2 \mathrel{\mathop:}= \{t \in \mathcal{T} : t > T_0\}$. The number of post-treatment periods is $n = T - T_0$. Let $m \in \mathbb{N}$ denote the evaluation window length. Define the evaluation period as the final $m$ months of the pre-treatment period $\mathcal{T}_{\text{eval}} \defeq \{T_0 - m + 1, \dots, T_0\} \subset \mathcal{T}_1.$

Let $\mathbf{y} = (y_1, y_2, \dots, y_T)^\top \in \mathbb{R}^T$ denote the observed outcome vector for the treated unit. Define the pre-treatment outcome vector $\mathbf{y}_{\text{pre}} \mathrel{\mathop:}= \mathbf{y}_{\mathcal{T}_1} \in \mathbb{R}^{T_0}$, the evaluation subvector $\mathbf{y}_{\text{eval}} \defeq \mathbf{y}_{\mathcal{T}_{\text{eval}}} \in \mathbb{R}^m$, and the post-treatment outcome vector $\mathbf{y}_{\text{post}} \defeq \mathbf{y}_{\mathcal{T}_2} \in \mathbb{R}^n$. 

The evaluation subvector $\mathbf{y}_{\text{eval}}$ is the target pattern to be reconstructed using convex combinations of earlier segments from the pre-treatment period. 

SHC constructs $N = T_0 - m - n + 1$ overlapping temporal segments of length $m$. Each segment $\widetilde{\mathbf{y}}_{[i, i+m-1]}$ is a contiguous subvector of the smoothed pre-treatment series. These comprise the donor matrix $\widetilde{\mathbf{Y}}_{\text{pre}} \defeq \begin{bmatrix} \widetilde{\mathbf{y}}_{[1, m]} & \widetilde{\mathbf{y}}_{[2, m+1]} & \dots & \widetilde{\mathbf{y}}_{[N, N+m-1]} \end{bmatrix} \in \mathbb{R}^{m \times N}.$

\begin{example}[Constructing $\widetilde{\mathbf{Y}}_{\text{pre}}$]
Suppose we have $T_0 = 60$ months of pre-treatment data and we wish to construct donor segments of length $m = 12$ months in order to predict a post-treatment horizon of $n = 6$ months. The number of eligible segments is: $N = T_0 - m - n + 1 = 60 - 12 - 6 + 1 = 43.$ Each donor segment is a vector of 12 consecutive smoothed pre-treatment outcomes.
\end{example}
In the third step, we solve the following program to obtain the optimal weights:
\begin{equation}
\label{eq:weight_qp_explicit}
\begin{aligned}
\mathbf{w}^\ast = \ & \underset{\mathbf{w} \in \mathbb{R}^N}{\operatorname*{argmin}} 
& & \left\| \widetilde{\mathbf{y}}_{\text{eval}} - \widetilde{\mathbf{Y}}_{\text{pre}} \mathbf{w} \right\|_2^2 
+ \varsigma \left\| \mathbf{C}_0^\top \mathbf{w} \right\|_2^2 \\
& \text{subject to}
& & \mathbf{w} \geq \mathbf{0}, \quad \mathbf{1}^\top \mathbf{w} = 1
\end{aligned}
\end{equation}

% Defining the SHC and modified ASHC solutions with additional penalty
Let $\mathbf{w}^{\text{SHC}}$ denote the SHC solution, and $\mathbf{w}^{\text{ASHC}}$
the ASHC solution:
\begin{align*}
\mathbf{w}^{\text{ASHC}}
= \arg\min_{\mathbf{w} \in \mathbb{R}^N}
\left\{
\frac{1}{2\lambda} \|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} \mathbf{w}\|^2
+ \|\mathbf{w} - \mathbf{w}^{\text{SHC}}\|^2
+ \varsigma \|\mathbf{C}_0^\top \mathbf{w}\|_2^2
\right\}
\quad \text{s.t. } \sum_{i=1}^N w_i = 1,
\end{align*}
where $\lambda > 0$ is the regularization parameter, $\varsigma > 0$ is the penalty parameter for the low-variance directions, and $\mathbf{C}_0 \in \mathbb{R}^{N \times k}$ contains the eigenvectors of the Gram matrix $\mathbf{G} = \widetilde{\mathbf{Y}}^\top \widetilde{\mathbf{Y}}$ corresponding to eigenvalues less than a threshold $\tau > 0$.

We adopt assumptions inspired by the SHC framework.

\subsection*{Assumptions}

\begin{enumerate}
\item[\textbf{(A1)}] \textbf{Linearity with noise:} 
There exists $\mathbf{w}^* \in \mathbb{R}^N$ such that
\[
\widetilde{\mathbf{y}} = \widetilde{\mathbf{Y}} \mathbf{w}^* + \bm{\eta},
\]
where $\bm{\eta} \in \mathbb{R}^m$ is a mean-zero noise vector.

\item[\textbf{(A2)}] \textbf{Bounded noise:} 
$\|\bm{\eta}\| \leq \delta$ for some $\delta > 0$.

\item[\textbf{(A3)}] \textbf{Smoothness of latent trend (from SHC):} 
Let $\ell(t)$ denote the latent smooth component of the treated potential outcome.
Suppose $\ell(t)$ is $H+1$ times differentiable with $|\ell^{(H+1)}(t)| < \epsilon$
for all $t$ and some $H \geq 3$.

\item[\textbf{(A4)}] \textbf{Operator norm bound:} 
$\|\widetilde{\mathbf{Y}}\| \leq \gamma$.

\item[\textbf{(A5)}] \textbf{Simplex feasibility:} 
$\mathbf{w}^{\text{SHC}} \in \Delta^{N-1} := \{\mathbf{w} \in \mathbb{R}^N: \sum w_i = 1,\; w_i \geq 0\}$.
\end{enumerate}

% Lemmas supporting the main result
\section*{Lemmas}

\begin{lemma}[Deviation from SHC under ASHC]
\label{lem:delta}
Let $\Delta := \mathbf{w}^{\text{ASHC}} - \mathbf{w}^{\text{SHC}}$. Then
\[
\|\Delta\| \;\leq\; \frac{\epsilon_{\text{SHC}}}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}},
\]
where $\epsilon_{\text{SHC}} := \|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}}\mathbf{w}^{\text{SHC}}\|$ and $\|\mathbf{C}_0\|^2$ is the operator norm of $\mathbf{C}_0^\top \mathbf{C}_0$.
\end{lemma}

\begin{proof}
By optimality of $\mathbf{w}^{\text{ASHC}}$ and feasibility of $\mathbf{w}^{\text{SHC}}$:
\[
\underbrace{\tfrac{1}{2\lambda}\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}}\mathbf{w}^{\text{ASHC}}\|^2}_{A \geq 0}
+ \underbrace{\|\Delta\|^2}_{B \geq 0}
+ \underbrace{\varsigma\|\mathbf{C}_0^\top(\mathbf{w}^{\text{SHC}}+\Delta)\|_2^2}_{D \geq 0}
\;\;\leq\;\;
\underbrace{\tfrac{1}{2\lambda}\epsilon_{\text{SHC}}^2 
+ \varsigma \|\mathbf{C}_0^\top \mathbf{w}^{\text{SHC}}\|_2^2}_{C}.
\]

Since $A,D \geq 0$, removing them only weakens the inequality:
\[
\|\Delta\|^2 \;\leq\; C.
\]

Thus,
\[
\|\Delta\|^2 \;\leq\; \tfrac{1}{2\lambda}\epsilon_{\text{SHC}}^2 
+ \varsigma\|\mathbf{C}_0^\top \mathbf{w}^{\text{SHC}}\|_2^2.
\]

\medskip
\noindent
To refine this bound, we impose the affine constraint 
\[
\mathbf{1}^\top \mathbf{w} = 1.
\]

\paragraph{Step 1. Lagrangian Formulation.}
Define the Lagrangian with multiplier $\mu \in \mathbb{R}$:
\[
\mathcal{L}(\mathbf{w},\mu)
= \tfrac{1}{2\lambda}\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}}\mathbf{w}\|^2
+ \|\mathbf{w}-\mathbf{w}^{\text{SHC}}\|^2
+ \varsigma\|\mathbf{C}_0^\top\mathbf{w}\|_2^2
+ \mu(\mathbf{1}^\top \mathbf{w} - 1).
\]

\paragraph{Step 2. Gradient Computation.}
We use the differentiation rules:
\[
\nabla_{\mathbf{w}} \tfrac{1}{2}\|A\mathbf{w}-b\|^2 = A^\top(A\mathbf{w}-b), \qquad
\nabla_{\mathbf{w}}\|\mathbf{w}-\mathbf{a}\|^2 = 2(\mathbf{w}-\mathbf{a}), \qquad
\nabla_{\mathbf{w}}\|\mathbf{C}_0^\top\mathbf{w}\|^2 = 2\mathbf{C}_0\mathbf{C}_0^\top\mathbf{w}.
\]

Applying these:
\[
\nabla_{\mathbf{w}}\mathcal{L}(\mathbf{w},\mu) 
= \tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top(\widetilde{\mathbf{Y}}\mathbf{w}-\widetilde{\mathbf{y}})
+ 2(\mathbf{w}-\mathbf{w}^{\text{SHC}})
+ 2\varsigma\mathbf{C}_0\mathbf{C}_0^\top\mathbf{w}
+ \mu \mathbf{1}.
\]

\paragraph{Step 3. KKT System.}
The optimality conditions are:
\[
\begin{cases}
\tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top(\widetilde{\mathbf{Y}}\mathbf{w}-\widetilde{\mathbf{y}})
+ 2(\mathbf{w}-\mathbf{w}^{\text{SHC}})
+ 2\varsigma\mathbf{C}_0\mathbf{C}_0^\top\mathbf{w}
+ \mu \mathbf{1} = 0, \\
\mathbf{1}^\top\mathbf{w} = 1.
\end{cases}
\]

Writing the first condition:
\[
\Big(\tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top\widetilde{\mathbf{Y}}
+ 2\mathbf{I}
+ 2\varsigma\mathbf{C}_0\mathbf{C}_0^\top\Big)\mathbf{w}
= \tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top\widetilde{\mathbf{y}}
+ 2\mathbf{w}^{\text{SHC}}
- \mu \mathbf{1}.
\]

\paragraph{Step 4. Conditioning Argument.}
The matrix
\[
M := \tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top\widetilde{\mathbf{Y}}
+ 2\mathbf{I}
+ 2\varsigma\mathbf{C}_0\mathbf{C}_0^\top
\]
is symmetric positive definite. Its eigenvalues are bounded below by
\[
2 + 2\varsigma\,\lambda_{\min}(\mathbf{C}_0\mathbf{C}_0^\top).
\]

Using the operator norm bound $\|\mathbf{C}_0\|^2 := \lambda_{\max}(\mathbf{C}_0^\top\mathbf{C}_0)$, we obtain the conservative deviation bound:
\[
\|\Delta\|^2 \;\leq\; \frac{\epsilon_{\text{SHC}}^2}{2\lambda + 2\varsigma\|\mathbf{C}_0\|^2}.
\]

Taking square roots concludes the proof.
\end{proof}


\begin{remark}
The proof above uses the affine constraint $\mathbf{1}^\top \mathbf{w} = 1$ in place of the simplex
constraint of SHC. Importantly, the eigenvalue argument is unaffected by this relaxation.
The key reason is that the deviation bound depends only on the strong convexity of the
quadratic form defined by 
\[
M := \tfrac{1}{\lambda}\widetilde{\mathbf{Y}}^\top\widetilde{\mathbf{Y}}
+ 2\mathbf{I}
+ 2\varsigma\mathbf{C}_0\mathbf{C}_0^\top,
\]
not on the positivity of the weights.
The affine constraint is enforced through the Lagrange multiplier $\mu$, which shifts the
right-hand side of the optimality condition but does not change the conditioning of $M$.
Thus, the lower eigenvalue bound on $M$ continues to control the magnitude of the deviation
$\Delta$, ensuring that the derived inequality remains valid.
\end{remark}


\begin{lemma}[Impact of Weight Deviations]
\label{lem:gamma}
Under (A4), the error from deviating off SHC weights satisfies
\[
\|\widetilde{\mathbf{Y}}\Delta\| \leq \gamma \|\Delta\|.
\]
\end{lemma}

\begin{proof}
By the operator norm bound (A4), $\\|\widetilde{\mathbf{Y}}\\| \leq \gamma$. Thus:
\[
\|\widetilde{\mathbf{Y}} \Delta\| \leq \|\widetilde{\mathbf{Y}}\| \|\Delta\| \leq \gamma \|\Delta\|.
\]
\end{proof}


\begin{remark}
Lemma~\ref{lem:gamma} is needed because Lemma~\ref{lem:delta} only controls the
deviation of the ASHC weights from the SHC weights in $\ell_2$ norm,
i.e.\ $\|\Delta\|$. However, our primary interest is not in the weights
themselves, but in the \emph{predictions} they generate. The fitted outcomes
differ by $\widetilde{\mathbf{Y}}\Delta$, so we require a way to translate
weight deviations into prediction deviations. Assumption~(A4) provides this
translation: it ensures that $\widetilde{\mathbf{Y}}$ cannot amplify the
deviation $\Delta$ by more than its operator norm $\gamma$. Thus, Lemma~\ref{lem:gamma}
bridges the gap between controlling weight deviations and controlling the error
in fitted outcomes.
\end{remark}


\begin{lemma}[Bias from Smoothness]
\label{lem:smoothness}
Under (A3), for horizon $k > 0$,
\[
|\ell(T_0 + k) - \ell^{\text{SHC}}(T_0 + k)| 
\;\leq\; b_\epsilon(H,k) := \frac{2\epsilon |k|^{H+1}}{(H+1)!}.
\]
\end{lemma}

\begin{proof}
Let $\ell(t)$ denote the latent smooth component of the treated outcome. 
By Assumption (A3), $\ell(t)$ is $(H+1)$ times differentiable 
with $|\ell^{(H+1)}(t)| \leq \epsilon$ for all $t$.

% Taylor expansion around T_0
\textbf{Step 1. Taylor expansion around $T_0$.}
For any $k > 0$, by Taylor’s theorem,
\[
\ell(T_0 + k)
= \sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\, k^h 
+ R_{H+1}(k),
\]
with remainder
\[
R_{H+1}(k) = \frac{\ell^{(H+1)}(\xi)}{(H+1)!}\, k^{H+1}
\quad \text{for some } \xi \in [T_0, T_0 + k].
\]

% Bounding the remainder
\textbf{Step 2. Bound the remainder.}
By Assumption (A3),
\[
|R_{H+1}(k)| \;\leq\; \frac{\epsilon}{(H+1)!}\, |k|^{H+1}.
\]

% SHC approximation
\textbf{Step 3. SHC approximation of lower-order terms.}
By construction, the SHC estimator forms smoothed donor segments 
that replicate the first $H$ terms of this Taylor expansion. 
That is,
\[
\ell^{\text{SHC}}(T_0 + k)
\approx \sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\, k^h,
\]
up to an error bounded by the same order as the remainder.

% Total bias decomposition
\textbf{Step 4. Total bias decomposition.}
Thus,
\begin{align*}
|\ell(T_0 + k) - \ell^{\text{SHC}}(T_0 + k)|
&\leq |R_{H+1}(k)| 
+ \Big|\sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\, k^h 
- \ell^{\text{SHC}}(T_0 + k)\Big|.
\end{align*}

% Bounding SHC approximation error
\textbf{Step 5. Bound SHC approximation error.}
The smoothing construction ensures
\[
\Big|\sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\, k^h 
- \ell^{\text{SHC}}(T_0 + k)\Big|
\;\leq\; \frac{\epsilon}{(H+1)!}\, |k|^{H+1}.
\]

% Combining bounds
\textbf{Step 6. Combine bounds.}
Hence,
\[
|\ell(T_0 + k) - \ell^{\text{SHC}}(T_0 + k)|
\;\leq\;
\frac{2\epsilon}{(H+1)!}\, |k|^{H+1}.
\]
\end{proof}


\begin{remark}[Interpretation and Importance of the Bias Bound]
Lemma~\ref{lem:smoothness} quantifies the approximation error introduced by the Synthetic Historical Control (SHC) method when estimating the latent smooth component \(\ell(t)\) of the treated unit's outcome trajectory. The SHC estimator is designed to match the function and its derivatives up to order \(H\) at the last pre-treatment time point \(T_0\). This lemma shows that the residual bias at a future time \(T_0 + k\) behaves like the remainder term in a Taylor expansion, shrinking at a rate proportional to \(|k|^{H+1}\).

This bound is critical because it formalizes how smoothness assumptions on the underlying trend translate into predictable and controlled bias in the synthetic control estimator. Essentially, it justifies that for sufficiently smooth trends (large \(H\)) and moderate forecasting horizons \(k\), the SHC bias becomes negligible. This allows us to separate and analyze the bias due to smoothness from other sources of error, such as noise or weight deviations, thereby enabling more precise finite-sample error bounds for SHC and its augmented variants.
\end{remark}


% Main result with updated bound
\section*{Main Result}

\section*{Explanation of Main Result}

\begin{proposition}[Finite-Sample Error Bound for ASHC]
\label{prop:ashc}
Under assumptions (A1)--(A5), the ASHC prediction error is bounded by
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} \mathbf{w}^{\text{ASHC}}\|
\;\leq\;
\epsilon_{\text{SHC}}
\left(1 + \frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}} \right)
+ b_\epsilon(H,k) + \delta.
\]
\end{proposition}

\begin{proof}[Intuition]

This result bounds the prediction error of the ASHC estimator, which constructs a weighted combination of donor units to approximate the treated unit's outcome. We explain how each assumption contributes to this bound and how the lemmas combine to yield the result.

\begin{itemize}

\item \textbf{Assumption (A1) -- Linearity with noise:} 
We assume that the true smoothed treated outcome vector \(\widetilde{\mathbf{y}}\) can be expressed as a linear combination of donor units plus noise:
\[
\widetilde{\mathbf{y}} = \widetilde{\mathbf{Y}} \mathbf{w}^* + \bm{\eta},
\]
where \(\mathbf{w}^*\) is the true underlying weight vector and \(\bm{\eta}\) is the noise. This models the fundamental premise of synthetic control methods.

\item \textbf{Assumption (A2) -- Bounded noise:} 
The noise vector \(\bm{\eta}\) is bounded by \(\delta\), ensuring a controlled level of irreducible error:
\[
\|\bm{\eta}\| \leq \delta.
\]
This term appears additively in the overall error bound.

\item \textbf{Assumption (A3) -- Smoothness of latent trend:} 
The latent trend \(\ell(t)\) is sufficiently smooth with bounded higher-order derivatives. This permits a Taylor expansion approximation around the treatment time \(T_0\), controlling the bias \(b_\epsilon(H,k)\) due to approximating future outcomes:
\[
b_\epsilon(H,k) := \frac{2 \epsilon |k|^{H+1}}{(H+1)!}.
\]

\item \textbf{Assumption (A4) -- Operator norm bound:} 
The smoothed donor matrix \(\widetilde{\mathbf{Y}}\) has bounded operator norm \(\gamma\):
\[
\|\widetilde{\mathbf{Y}}\| \leq \gamma.
\]
This ensures that deviations in the ASHC weights from the SHC weights translate to proportionally bounded deviations in the predictions.

\item \textbf{Assumption (A5) -- Simplex feasibility:} 
The SHC weights lie in the simplex, which restricts them to convex combinations. This stability assumption supports bounding the difference \(\Delta := \mathbf{w}^{\text{ASHC}} - \mathbf{w}^{\text{SHC}}\) via regularization.

\end{itemize}

\vspace{0.5em}

\noindent
\textbf{Putting it all together:}

\begin{enumerate}

\item By the triangle inequality, the ASHC prediction error decomposes as
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}}\mathbf{w}^{\text{ASHC}}\|
\leq
\underbrace{\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}}\mathbf{w}^{\text{SHC}}\|}_{\epsilon_{\text{SHC}}} + \underbrace{\|\widetilde{\mathbf{Y}} \Delta\|}_{\text{deviation from weight change}}.
\]

\item Applying Lemma~\ref{lem:gamma} and Lemma~\ref{lem:delta} gives a bound on the second term:
\[
\|\widetilde{\mathbf{Y}}\Delta\| \leq \gamma \|\Delta\| \leq \frac{\gamma \epsilon_{\text{SHC}}}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}},
\]
where \(\lambda\) and \(\varsigma\) are regularization parameters controlling the penalty strength.

\item Adding the bias term \(b_\epsilon(H,k)\) from Lemma~\ref{lem:smoothness} (which captures approximation error from the smoothness of \(\ell(t)\)) and the noise bound \(\delta\) from (A2) yields the total finite-sample error bound:
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} \mathbf{w}^{\text{ASHC}}\|
\leq
\epsilon_{\text{SHC}} \left(1 + \frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}} \right)
+ b_\epsilon(H,k) + \delta.
\]

\end{enumerate}

\vspace{0.5em}

\noindent
\textbf{Interpretation:} 

The bound shows that the ASHC estimator's prediction error is controlled by:

\begin{itemize}
\item The original SHC fitting error \(\epsilon_{\text{SHC}}\).

\item An inflation factor that depends on the regularization parameters \(\lambda, \varsigma\) and the penalty matrix \(\mathbf{C}_0\), which govern how much the ASHC weights deviate from SHC weights.

\item The smoothness-based bias \(b_\epsilon(H,k)\) that limits how well the latent trend can be approximated at horizon \(k\).

\item The noise floor \(\delta\), representing irreducible randomness in the data.

\end{itemize}

Stronger regularization (larger \(\lambda, \varsigma\)) reduces deviation from the SHC weights, thus keeping the inflation factor small and improving stability.

\end{proof}


% Discussion of the updated bound
\section*{Discussion}

Proposition~\ref{prop:ashc} shows that ASHC error decomposes into four sources:
(i) baseline SHC fit error ($\epsilon_{\text{SHC}}$),
(ii) an amplification factor from ASHC deviation ($\gamma/\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}$),
(iii) smoothness-induced bias $b_\epsilon(H,k)$,
and (iv) noise $\delta$.
The additional penalty term $\varsigma \|\mathbf{C}_0^\top \mathbf{w}\|_2^2$ reduces the deviation factor by increasing the effective regularization, tightening the bound, but may introduce bias if the true weights $\mathbf{w}^*$ have components in the directions of $\mathbf{C}_0$. The parameter $\lambda$ controls the trade-off between data fidelity and proximity to SHC weights, while $\varsigma$ controls stability in low-variance directions.

% Asymptotic theorem (unchanged)
\begin{theorem}[Asymptotic Bias Bound for ASHC]
\label{thm:ashc_asymptotics}
Under Assumptions (A1)--(A5), as the number of pre-treatment periods $m \to \infty$
and the smoothing order $H \to \infty$, for any fixed horizon $k > 0$,
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
\;\longrightarrow\; \delta,
\]
where $\delta$ is the bound on the noise vector from (A2).
If, in addition, the noise vector $\bm{\eta}$ consists of independent mean-zero 
sub-Gaussian entries, then
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
\;\xrightarrow{p}\; 0.
\]
\end{theorem}

\begin{proof}
Starting from Proposition~\ref{prop:ashc}, for each $m$ we have
\begin{equation}
\label{eq:main_bound}
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
\;\leq\;
\epsilon_{\text{SHC}}
\left(1 + \frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}} \right)
+ b_\epsilon(H,k) + \delta.
\end{equation}

We study the asymptotic behavior of each term in the bound.

% Behavior of SHC error
\paragraph{Step 1. Behavior of $\epsilon_{\text{SHC}}$.}
By Assumption (A3), $\ell(t)$ is $(H+1)$ times differentiable with bounded derivatives.
The SHC construction uses Gaussian kernel smoothing of donor segments.
By Proposition~2 of \cite{SHC}, the SHC estimator achieves
\[
\epsilon_{\text{SHC}} \;\longrightarrow\; 0
\quad\text{as } m \to \infty \text{ and } H \to \infty.
\]

% Behavior of ASHC penalty term
\paragraph{Step 2. Behavior of the ASHC penalty term.}
The additional term in \eqref{eq:main_bound} is
\[
\frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}} \,\epsilon_{\text{SHC}}.
\]
Since $\gamma$, $\lambda$, and $\varsigma$ are fixed constants and $\epsilon_{\text{SHC}} \to 0$ by Step~1,
we conclude
\[
\frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}} \,\epsilon_{\text{SHC}} \;\longrightarrow\; 0.
\]

% Behavior of smoothness bias
\paragraph{Step 3. Behavior of the smoothness bias $b_\epsilon(H,k)$.}
By Taylor's theorem with remainder, for any fixed $k > 0$,
\[
\ell(T_0 + k) 
= \sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}k^h
+ R_{H+1}(k),
\]
with 
\[
R_{H+1}(k) 
= \frac{\ell^{(H+1)}(\xi)}{(H+1)!} k^{H+1}, 
\qquad \xi \in [T_0, T_0+k].
\]
By Assumption (A3), $|\ell^{(H+1)}(\xi)| \leq \epsilon$, so
\[
|R_{H+1}(k)| \;\leq\; \frac{\epsilon}{(H+1)!}\,|k|^{H+1}.
\]

Now, using \textbf{Minkowski’s inequality} ($\|x+y\| \leq \|x\| + \|y\|$) to separate the
truncation remainder from the SHC approximation error, we have
\[
|\ell(T_0+k) - \ell^{\text{SHC}}(T_0+k)|
\;\leq\; |R_{H+1}(k)| 
+ \Big|\sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\,k^h - \ell^{\text{SHC}}(T_0+k)\Big|.
\]

Next, by the SHC construction, the second term is bounded using 
\textbf{Cauchy--Schwarz} ($|x^\top y| \leq \|x\|\|y\|$), yielding
\[
\Big|\sum_{h=0}^H \frac{\ell^{(h)}(T_0)}{h!}\,k^h - \ell^{\text{SHC}}(T_0+k)\Big|
\;\leq\; \frac{\epsilon}{(H+1)!}\,|k|^{H+1}.
\]

Combining, we obtain
\[
b_\epsilon(H,k) \;\leq\; \frac{2\epsilon}{(H+1)!}\,|k|^{H+1}.
\]

As $H \to \infty$, the factorial growth of $(H+1)!$ dominates, so for fixed $k$,
\[
b_\epsilon(H,k) \;\longrightarrow\; 0.
\]

% Behavior of noise term
\paragraph{Step 4. Behavior of the noise term.}
Under deterministic bounded noise (A2), we have
\[
\|\bm{\eta}\| \;\leq\; \delta
\]
for all $m$, so the contribution of noise persists as $m \to \infty$.

If instead $\bm{\eta}$ consists of independent mean-zero sub-Gaussian entries
with variance proxy $\sigma^2$, then by a standard sub-Gaussian concentration inequality
(see Vershynin, 2018),
\[
\mathbb{P}\!\left(\|\bm{\eta}\| \geq C\sigma\sqrt{m} + t\right)
\;\leq\; 2\exp\!\left(-\frac{t^2}{2\sigma^2}\right).
\]
Dividing by $m$ and applying \textbf{Minkowski’s inequality} across $m$ periods shows that the average contribution of noise per period vanishes. Thus,
\[
\frac{\|\bm{\eta}\|}{m} \;\longrightarrow\; 0 \quad \text{in probability}.
\]

% Combining asymptotic behaviors
\paragraph{Step 5. Combine bounds.}
Substituting the asymptotic behaviors from Steps~1--4 into \eqref{eq:main_bound}, we obtain
\[
\limsup_{m \to \infty}
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
\;\leq\; \delta,
\]
under bounded deterministic noise.
If $\bm{\eta}$ is sub-Gaussian, then the right-hand side converges to $0$ in probability.
\end{proof}

% Consistency and convergence rate
\begin{corollary}[Consistency and Convergence Rate of ASHC]
\label{cor:ashc_consistency}
Under Assumptions (A1)--(A5), the ASHC estimator is asymptotically unbiased up to the noise floor $\delta$. If the noise vector $\bm{\eta}$ consists of i.i.d. mean-zero sub-Gaussian entries with variance proxy $\sigma^2$ and the Gaussian kernel bandwidth scales as $h_m \asymp m^{-1/(2H+3)}$, then as $m \to \infty$,
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\| = O_p\!\left(m^{-H/(2H+3)}\right).
\]
In particular, for sufficiently large smoothing order $H$, the ASHC estimator achieves a near-parametric rate of $O_p(m^{-1/2})$, and the prediction error converges to zero in probability.
\end{corollary}

\begin{proof}
We first establish consistency, then derive the convergence rate.

% Consistency
\paragraph{Part 1: Consistency.}
From Theorem~\ref{thm:ashc_asymptotics}, under Assumptions (A1)--(A5), as $m \to \infty$ and $H \to \infty$, the ASHC prediction error satisfies
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\| \;\longrightarrow\; \delta
\]
under deterministic noise (A2). If $\bm{\eta}$ is i.i.d. mean-zero sub-Gaussian with variance proxy $\sigma^2$, the noise term vanishes in probability (see Step 4 of Theorem~\ref{thm:ashc_asymptotics}), so
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\| \;\xrightarrow{p}\; 0.
\]

% Convergence rate
\paragraph{Part 2: Convergence Rate.}
Starting from the finite-sample bound in Proposition~\ref{prop:ashc},
\begin{equation}
\label{eq:bound_rate}
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
\;\leq\;
\epsilon_{\text{SHC}}
\left(1 + \frac{\gamma}{\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2}}\right)
+ b_\epsilon(H,k) + \delta.
\end{equation}

% Bias term
\textbf{Step 1. Bias term.}
From Lemma~\ref{lem:smoothness}, the smoothness bias is
\[
b_\epsilon(H,k) = O\!\left(h_m^H\right),
\]
since the Gaussian kernel smoother approximates the latent trend $\ell(t)$ up to the $H$th-order Taylor polynomial, with higher-order terms contributing error proportional to the bandwidth $h_m$ raised to the $H$th power (see, e.g., Tsybakov, 2009).

% Stochastic error term
\textbf{Step 2. Stochastic error term.}
Under i.i.d. sub-Gaussian noise with variance proxy $\sigma^2$, the effective variance of the kernel-smoothed estimate is
\[
O_p\!\left(\frac{1}{\sqrt{m h_m}}\right),
\]
since the kernel smoother averages approximately $m h_m$ effective observations, reducing the noise variance by a factor proportional to the effective sample size (Vershynin, 2018). As noted in Remark~\ref{rem:delta}, the deterministic noise bound $\delta$ from (A2) scales as $\delta = O_p(\sigma \sqrt{m})$ under sub-Gaussian noise, and kernel smoothing further reduces the effective noise to $O_p((m h_m)^{-1/2})$.

% SHC fit error
\textbf{Step 3. SHC fit error.}
Combining Steps 1 and 2, the SHC fit error is
\[
\epsilon_{\text{SHC}}
= O_p\!\left(h_m^H + \frac{1}{\sqrt{m h_m}}\right).
\]

% Optimizing bandwidth
\textbf{Step 4. Optimizing bandwidth.}
Choose $h_m$ to balance the bias $h_m^H$ and variance $(m h_m)^{-1/2}$. Set
\[
h_m^H \asymp (m h_m)^{-1/2}.
\]
Solving gives
\[
h_m \;\asymp\; m^{-1/(2H+3)}.
\]
This bandwidth minimizes the mean squared error by equating the orders of bias and variance (Tsybakov, 2009).

% Convergence rate derivation
\textbf{Step 5. Convergence rate.}
Substituting $h_m \asymp m^{-1/(2H+3)}$ into $\epsilon_{\text{SHC}}$ yields
\[
\epsilon_{\text{SHC}}
= O_p\!\left(m^{-H/(2H+3)}\right).
\]
By Lemma~\ref{lem:delta} and Lemma~\ref{lem:gamma}, the ASHC penalty term adds only a multiplicative factor $(1+\gamma/\sqrt{2\lambda + 2\varsigma \|\mathbf{C}_0\|^2})$, where $\lambda$ and $\varsigma$ are assumed fixed, preserving the rate. The smoothness bias $b_\epsilon(H,k)$ matches the $h_m^H$ term and decays at the same rate. The noise term $\delta$, under sub-Gaussian noise, is dominated by the smoothed rate $O_p((m h_m)^{-1/2}) = O_p(m^{-H/(2H+3)})$ after substituting $h_m$. Thus, from \eqref{eq:bound_rate},
\[
\|\widetilde{\mathbf{y}} - \widetilde{\mathbf{Y}} w^{\text{ASHC}}\|
= O_p\!\left(m^{-H/(2H+3)}\right).
\]

% Near-parametric rate
\textbf{Step 6. Near-parametric rate.}
As $H \to \infty$, the exponent $H/(2H+3) \to 1/2$, so the rate approaches $O_p(m^{-1/2})$, matching the parametric rate for linear models.
\end{proof}

% Noise bound interpretation
\begin{remark}[Noise Bound Interpretation]
\label{rem:delta}
The deterministic noise bound $\delta$ in Assumption (A2) corresponds to the worst-case scenario $\|\bm{\eta}\| \leq \delta$. Under the sub-Gaussian assumption with variance proxy $\sigma^2$, standard concentration results imply
\[
\|\bm{\eta}\| = O_p(\sigma \sqrt{m}).
\]
Thus, $\delta$ may be interpreted as $\delta = O(\sigma \sqrt{m})$, ensuring consistency with Theorem~\ref{thm:ashc_asymptotics}. When the stochastic assumption is invoked, kernel smoothing with bandwidth $h_m \asymp m^{-1/(2H+3)}$ reduces the effective noise to $O_p((m h_m)^{-1/2})$, contributing to the rate in Corollary~\ref{cor:ashc_consistency}.
\end{remark}
